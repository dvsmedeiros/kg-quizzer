import re

def extract_school_name(url):
  """
  Extracts the school name from a URL.

  Args:
    url: The URL of the high school.

  Returns:
    The school name, or None if the school name could not be extracted.
  """
  match = re.search(r"/([^/]+)$", url)
  if match:
    return match.group(1).replace("_", " ").title()
  else:
    return None
